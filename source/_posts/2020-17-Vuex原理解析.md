## 原理
[Vuex解读](http://www.gaozhenchao.cn/posts/202008/)，vuex的基本用法可以看这篇文章。这里主要分析实现思路。

首先我们要明白vuex是一个插件，是插件就要实现install方法，其次，我们的Store内部需要实现一个commit方法用来改变state的值，同时还需要实现dispatch方法来调用commit改变state。而且我们的state要是响应式的数据，所以我们可以new一个Vue在里面声明data来存储用户传入的state。这样我们的state更新就可以调用render方法更改页面中用到state的数据。这就是我们的基本思路。

### 实现响应式state
```js
let Vue;

class Store {
  constructor(options){
    this.$options = options
    // 响应化处理state
    this.state = new Vue({
      data: options.state
    })
  }
}

function install(_Vue){
  Vue = _Vue;

  // 混入$store
  Vue.mixin({
    beforeCreate(){
      if(this.$options.store){
        Vue.prototype.$store = this.$options.store
      }
    }
  })
}

export default {
  Store,
  install
}
```
这样我们就创建了一个响应式的state了。
### 完成commit与dispatch方法
```js
let Vue;

class Store {
  constructor(options) {
    this.$options = options
    this._mutations = options.mutataions;
    this._actions = options.actions;

    // 响应化处理state
    this.state = new Vue({
      data: options.state
    })

    // actions中异步操作比如设置定时器会报错，因为this指向的是window
    // 绑定commit，dispatch的上下文为store实例
    this.commit = this.commit.bind(this)
    this.dispatch = this.dispatch.bind(this)
  }

  // store.commit('add', 1)
  // type: mutation的类型
  // payload: 载荷，是参数
  commit(type, payload) {
    const entry = this._mutations[type];
    if (entry) {
      entry(this.state, payload)
    }
  }

  dispatch(type, payload) {
    const entry = this._actions[type];
    if (entry) {
      entry(this, payload)
    }
  }
}

function install(_Vue) {
  Vue = _Vue;

  // 混入$store
  Vue.mixin({
    beforeCreate() {
      if (this.$options.store) {
        Vue.prototype.$store = this.$options.store
      }
    }
  })
}

export default {
  Store,
  install
}
```

### 完成getters
```js
let Vue;

class Store {
  constructor(options) {
    this.$options = options;
    this._mutations = options.mutataions;
    this._actions = options.actions;
    this._wrappedGetters = options.getters;

    // 定义computed选项
    const computed = {};
    this.getters = {};

    // {doubleCounter(state){}}
    const store = this;
    Object.keys(this._wrappedGetters).forEach(key => {
      // 获取用户定义的getter
      const fn = store._wrappedGetters[key];
      // 转换为computed可以使用的无参数形式
      computed[key] = function() {
        return fn(store.state)
      }
      // 给getters定义只读属性
      Object.defineProperty(store.getters, key, {
        get: () => store._vm[key]
      })
    })

    响应化处理state
    this.state = new Vue({
      data: options.state
    })

    // actions中异步操作比如设置定时器会报错，因为this指向的是window
    // 绑定commit，dispatch的上下文为store实例

    this.dispatch = this.dispatch.bind(this)
  }

  // type: mutation的类型
  // payload: 载荷，是参数

      entry(this.state, payload)
    }
  }

    const entry = this._actions[type];

      entry(this, payload)
    }
  }
}

```
简单说明一下，我们常说getters就相当于vue中的计算属性computed，所以这里我们也使用computed来实现getters，因为computed是一个没有参数的函数，所以我们创建一个高阶函数，在computed里面给getters的函数传入参数并执行我们真正想执行它。

### 调整state
到目前为止，基本的工作我们就全部完成了，但你可能发现了一个问题，我们的state现在是谁都可以修改的，这与实际vuex中的state不符，所以我们要想办法给保护一下state。下面是完整代码。
```js
let Vue;

class Store {
  constructor(options) {
    this.$options = options;
    this._mutations = options.mutataions;
    this._actions = options.actions;
    this._wrappedGetters = options.getters;

    // 定义computed选项
    const computed = {};
    this.getters = {};

    // {doubleCounter(state){}}
    const store = this;
    Object.keys(this._wrappedGetters).forEach(key => {
      // 获取用户定义的getter
      const fn = store._wrappedGetters[key];
      // 转换为computed可以使用的无参数形式
      computed[key] = function() {
        return fn(store.state)
      }
      // 给getters定义只读属性
      Object.defineProperty(store.getters, key, {
        get: () => store._vm[key]
      })
    })
    // 响应化处理state
    // this.state = new Vue({
    //   data: options.state
    // })
    this._vm = new Vue({
      data: {
        // 加两个$，Vue不会做代理，通过vm无法访问到$$state
        $$state: options.state
      }
    })
    // actions中异步操作比如设置定时器会报错，因为this指向的是window
    // 绑定commit，dispatch的上下文为store实例
    this.commit = this.commit.bind(this)
    this.dispatch = this.dispatch.bind(this)
  }
  // 存取器，store.state
  get state(){
    return this._vm._data.$$state
  }

  set state(){
    console.error("?你在想pichi");
  }

  // store.commit('add', 1)
  // type: mutation的类型
  // payload: 载荷，是参数
  commit(type, payload) {
    const entry = this._mutations[type];
    if (entry) {
      entry(this.state, payload)
    }
  }
  
  dispatch(type, payload){
    const entry = this._actions[type];
    if(entry){
      entry(this, payload)
    }
  }
}

function install(_Vue) {
  Vue = _Vue;

  // 混入$store
  Vue.mixin({
    beforeCreate() {
      if (this.$options.store) {
        Vue.prototype.$store = this.$options.store

    }
  })
}

export default {
  Store,
  install
}
```
